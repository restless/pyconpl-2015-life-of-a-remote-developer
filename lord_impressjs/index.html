    default='django.core.mail.backends.smtp.EmailBackend')

    'default': env.db("DATABASE_URL",
    default="postgres://localhost/{{cookiecutter.repo_name}}"),
}

Plays and Playbooks

File: webservers.yml

- name: Provision a {{ application_name }} web server
 hosts: webservers
 remote_user: root
 vars:
     - update_apt_cache: yes

 tasks:
     - name: Create deploy user
       user: name={{ deploy_user }} shell=/bin/bash
                               generate_ssh_key=yes
                                  ssh_key_bits=2048
                           ssh_key_file=.ssh/id_rsa

Plays and Playbooks 2

File: webservers.yml

- name: Install Nginx
  apt: name=nginx update_cache={{ update_apt_cache }} state=installed
  tags: packages
- name: Create the Nginx configuration file
  template: src=nginx_site_config.j2
            dest=/etc/nginx/sites-available/{{ application_name }}
            backup=yes
  notify: reload nginx
roles:
  - { role: gdamjan.uwsgi }

Ansistrano flow

before-setup.yml

- name: Turn maintenance mode on
  file: src={{ ansistrano_deploy_to }}/shared/maintenance.html
        dest={{ ansistrano_deploy_to }}/shared/maintenance_on.html
        owner={{ deploy_user }}
        group=www-data
        state=link

before-symlink.yml

- name: Install web app dependencies
  pip: requirements={{ ansistrano_release_path.stdout }}/requirements/{{env}}.txt
    virtualenv={{virtualenv_dir}}
    #virtualenv_command=virtualenv-3.4

- name: Django migrate
  django_manage: command=migrate
    app_path={{ ansistrano_release_path.stdout }}/app
    settings={{django_settings_module}}
    virtualenv={{virtualenv_dir}}

- name: Django collectstatic
  django_manage: command=collectstatic
    app_path={{ ansistrano_release_path.stdout }}/app
    settings={{django_settings_module}}
    virtualenv={{virtualenv_dir}}

after-symlink.yml

- name: Reload app vassal
  file: path=/etc/uwsgi/vassals/{{ application_name }}.ini
        state=touch

- name: Turn maintenance mode off
  file: path={{ ansistrano_deploy_to }}/shared/maintenance_on.html
        state=absent
